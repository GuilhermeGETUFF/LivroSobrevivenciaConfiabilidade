# Seleciona as colunas para exibição
tabela <- cronograma %>%
  select(Data, Conteúdo)

# Cria a tabela inicial
kbl_out <- kbl(tabela, align = "l", col.names = c("Data", "Conteúdo")) %>%
  kable_styling(full_width = FALSE, bootstrap_options = c("striped", "hover")) %>%
  row_spec(0, bold = TRUE)

# Aplica a cor de fundo linha a linha
for (i in 1:nrow(tabela)) {
  kbl_out <- kbl_out %>%
    row_spec(i, background = cronograma$Cor[i])
}

# Imprime a tabela
